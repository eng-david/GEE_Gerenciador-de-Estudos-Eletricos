CREATE TABLE sqlite_sequence(name,seq);

CREATE TABLE users (
	id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	username TEXT UNIQUE NOT NULL,
	hash TEXT NOT NULL
);

CREATE TABLE customers (
	id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	user_id INTEGER,
	createddate TEXT,
	modifieddate TEXT,
	name TEXT,
	email TEXT,
    phone TEXT,
    zipcode NUMERIC,
	state TEXT,
	city TEXT,
	street TEXT,
	district TEXT,
	number NUMERIC,
	complement TEXT
);

CREATE TABLE projects (
	id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	user_id INTEGER,
	createddate TEXT,
	modifieddate TEXT,
	name TEXT,
	customer_id NUMERIC,
	profile_id NUMERIC,
	tri_sc NUMERIC,
    bi_sc NUMERIC,
    lg_sc_max NUMERIC,
	lg_sc_min NUMERIC,
	voltage NUMERIC,
	demand NUMERIC,
	pf NUMERIC,
	tr_qty NUMERIC
);

CREATE TABLE profiles (
	id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	user_id INTEGER,
	createddate TEXT,
	modifieddate TEXT,
	name TEXT,
	threshold_factor NUMERIC,
	n_factor_threshold NUMERIC,
	max_n_threshold NUMERIC,
	inansi_factor NUMERIC,
	iansi_time NUMERIC,
	mag_oil_factor NUMERIC,
	mag_dry_factor NUMERIC,
	trip_factor NUMERIC,
	n_trip_factor NUMERIC,
	delay_50 NUMERIC
);

CREATE TABLE transformers (
	id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	project_id NUMERIC,
	power_kva NUMERIC,
	impedance NUMERIC,
	type NUMERIC
);
